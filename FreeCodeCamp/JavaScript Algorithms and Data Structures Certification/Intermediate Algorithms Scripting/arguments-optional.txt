function addTogether(...args) {
	if (args.length > 1 && args.every(item => typeof item === "number")) {
		// console.log("Is this is a fucking array");
		return args.reduce((a, b) => a + b);
	}

	if (!(args.every(item => typeof item === "number"))) {
		return undefined;
	}

	return function(x) {
		if (typeof x !== "number") {
			return undefined;
		}
		return args[0]+x;
	}
	
}


addTogether(2,3);
