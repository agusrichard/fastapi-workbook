function pairElement(str) {

	let outArr = []
	for (let i = 0; i < str.length; i++) {
		var inArr = []
		inArr.push(str[i]);
		switch (str[i]) {
			case "A":
				inArr.push("T");
				break;
			case "T":
				inArr.push("A");
				break;
			case "G":
				inArr.push("C");
				break;
			case "C":
				inArr.push("G");
		}
		outArr.push(inArr);
	}
	return outArr;
}

pairElement("GCG");
