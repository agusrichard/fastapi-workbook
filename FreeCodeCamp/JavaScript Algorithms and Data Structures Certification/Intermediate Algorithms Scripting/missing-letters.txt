function fearNotLetter(str) {
	var alphabet = "abcdefghijklmnopqrstuvwxyz";
	var trueStr = alphabet.slice(alphabet.indexOf(str[0]), alphabet.indexOf(str[0])+str.length+1);
	for (let i = 0; i < trueStr.length; i++) {
		if (str.indexOf(trueStr[i]) < 0) {
			return trueStr[i]
		}
	}
	return undefined;

}


fearNotLetter("abce");
