function translatePigLatin(str) {
	// console.log(/^[aeiou]/.test())
	// console.log(/[aeiou]/.test(str));
	if (!/[aeiou]/.test(str)) {
		return str + "ay";
	} else if (/^[aeiou]/.test(str)) {
		return str + "way";
	} else if (/([^aeiou]+)/.test(str)) {
    return str.replace(/([^aeiou]+)([aeiou][a-z]+)/, "$2$1" + "ay")
	}
  	
}

translatePigLatin("consonant");
