function sumAll(arr) {

      if (arr[0] < arr[1]) {
            var total = 0;
            for (let i = arr[0]; i <= arr[1]; i++) {
                  total += i;
                  console.log(total);
            }
      } else {
            var total = 0;
            for (let i = arr[1]; i <= arr[0]; i++) {
                  total += i;
                  console.log(total);
            }
      }
      return total;
}

sumAll([1, 4]);
